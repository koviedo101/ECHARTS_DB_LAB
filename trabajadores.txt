CREATE TABLE `trabajadores` (
   `dpi` varchar(500) DEFAULT NULL,
   `id` int NOT NULL AUTO_INCREMENT,
   `nombre` varchar(500) DEFAULT NULL,
   `nit` varchar(500) DEFAULT NULL,
   `puesto` varchar(500) DEFAULT NULL,
   `password` varchar(500) DEFAULT NULL,
   `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
   `rol` int DEFAULT NULL,
   `empresa` int DEFAULT NULL,
   `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
   `sueldo` decimal(10,2) DEFAULT '1.00',
   `telefono` varchar(250) DEFAULT '',
   `multiplicador_extra` decimal(5,3) NOT NULL DEFAULT '1.000',
   `titulo` varchar(15) DEFAULT NULL,
   `deleted_at` timestamp NULL DEFAULT NULL,
   `sueldo_mensual` decimal(10,2) NOT NULL DEFAULT '3500.00',
   `bonificacion` decimal(10,2) DEFAULT '250.00',
   PRIMARY KEY (`id`),
   KEY `rol` (`rol`),
   KEY `empresa` (`empresa`),
   CONSTRAINT `trabajadores_ibfk_1` FOREIGN KEY (`rol`) REFERENCES `roles` (`id`),
   CONSTRAINT `trabajadores_ibfk_2` FOREIGN KEY (`empresa`) REFERENCES `companies` (`id`)
 ) ENGINE=InnoDB AUTO_INCREMENT=30 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
